00118 {
00119     int RetState = 0;
00120 
00121 //==ENABLE LOG===============================================================//

00132 
00133 //==INITIALIZE INTERFACE=====================================================//

00144 //==READ FW and HW VERSIONS==================================================//

00146         StdUseCase_ReadFWVersion();

00153 
00154 //==GET DEVICE UNDER JTAG CONTROL============================================//
00155 
00156         StdUseCase_GetDevice(&g_sOpt);
00157 
00158         // if Breakpoints should be set
00159         if(sOpt->pcBreakpoints != NULL)
00160         {
00161             StdUseCase_EemInit();
00162         }
00163 
00164 //==ERASE MEMORY (optional)==================================================//
00165 
00166         StdUseCase_Erase(&g_sOpt);
00167 
00168 //==PROGRAM & VERIFY / READ OUT DEVICE MEMORY================================//
00169 
00170         StdUseCase_ReadWrite(&g_sOpt, &g_RTE);

00174         StdUseCase_Reset(PUC_RESET, FALSE, FALSE);
00175 
00176 //==SET BREAKPOINTS==========================================================//
00177

00184 //==CHECK IF DEVICE SHOULD BE SECURED========================================//

00189 //==DISPLAY INITIALIZATION RESULTS===========================================//

00191         StdUseCase_DisplayResults(argc, argv, &g_sOpt);
00192 
00193 //==RUN TO BREAKPOINT(S)=====================================================//
00194
