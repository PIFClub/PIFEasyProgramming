01021 int ReadSpacedArgs(int argc, char* argv[], int* p_optind, unsigned short specs, char* processed)
01022 {
01023     char* optarg;               // pointer to first part of argument
01024     int optind = *p_optind;     // argument index
01025     int runtime_end;            // index to last part of argument
01026     
01027     if (specs & IS_ATTACHED)    // if argument is attached to trigger
01028     {
01029         // first part of argument follows trigger 
01030         optarg = &argv[optind][2];
01031     }
01032     else
01033     {
01034         // first part of argument is the following input argument (optind++ externally)
01035         optarg = argv[optind];
01036     }
01037     
01038     if (!(specs & IS_FILENAME))    // if argument is of type '[]' specs
01039     {                                                                        
01040         if (*optarg != '[')    // if the argument doesn't begin with '['
01041         {
01042             Print_N_Log(ERR_MSG,"* Error: '[' expected at beginning of\n");
01043             Print_N_Log(ERR_MSG,"* exit_spec list -- z\n");    
01044             return('?');
01045         }
01046     }
01047     strcpy(processed, optarg);    // copy first argument into processed
01048     
01049     if (!(specs & IS_FILENAME))    // if argument is of type '[]' specs
01050     {
01051         for (runtime_end = optind++; runtime_end < argc; runtime_end++)    // look through all following input args                            
01052         {                                                                  //optind++ (first part read already)                        
01053             if (strchr(argv[runtime_end], ']') != NULL )    // until ']' is reached, then:
01054             {
01055                 break;    // stop counting
01056             }
01057         }                                                                    
01058         if (runtime_end >= argc)    // if ']' is not found
01059         {
01060             Print_N_Log(ERR_MSG,"* Error: ']' expected at end of\n");
01061             Print_N_Log(ERR_MSG,"* exit_spec list -- z\n\n");        
01062             return('?');
01063         }   // runtime_end indexes last part of arg                                                                    
01064     }    
01065     else    // else if argument is of type filename
01066     {
01067         for (runtime_end = optind++; runtime_end < argc; runtime_end++)    // search for an extension until found                            
01068         {                                                                                            
01069             if(    ( strstr(argv[runtime_end], ".txt") != NULL ) || ( strstr(argv[runtime_end], ".TXT") != NULL ) ||
01070                    ( strstr(argv[runtime_end], ".a43") != NULL ) || ( strstr(argv[runtime_end], ".A43") != NULL ) ||
01071                    ( strstr(argv[runtime_end], ".hex") != NULL ) || ( strstr(argv[runtime_end], ".HEX") != NULL ) ||
01072                    ( strstr(argv[runtime_end], ".int") != NULL ) || ( strstr(argv[runtime_end], ".INT") != NULL ))
01073             {
01074                 break;
01075             }
01076         }
01077         if (runtime_end >= argc)    // if not found
01078         {
01079             Print_N_Log(ERR_MSG,"* file extension unknown -- w\n\n");
01080             return('?');
01081         }   // runtime_end indexes last part of arg
01082     }
01083     
01084     if (specs & REMOVE_SPACES)    // if remove ' ' is set
01085     {
01086         while (optind <= runtime_end)    // while unread parts exist
01087         {
01088             strcat(processed, argv[optind++]);    // add to processed, removing ' '
01089         }
01090     }
01091     else
01092     {
01093         while (optind <= runtime_end)    // while unread parts exist
01094         {
01095             *(strchr(processed, '\0')+1) = '\0';    // add to processed
01096             *strchr(processed, '\0') = ' ';         // w/o removing ' '
01097             strcat(processed, argv[optind++]);                                
01098         }
01099     }
01100     
01101     *p_optind = optind;    // update value of external optind
01102     return(0);

01112 STATUS_T ParseInputFile(char targetcode[], LONG* lStartAddress, LONG* lProgramLength)
01113 {
01114     char* buffer = NULL;
01115     LONG lSize = 0;
01116     size_t result;
01117 
01118     WORD start = 0;
01119     WORD length = 0;
01120 
01121     FILE* lInputFile = NULL;
01122 
01123     if((lInputFile = fopen(targetcode, "r")) == NULL)
01124     {
01125         Print_N_Log(ERR_MSG,"* Failed to open input file!\n");
01126         return STATUS_ERROR;
01127     }
01128     else
01129     {
01130         fseek(lInputFile , 0 , SEEK_END);
01131         lSize = ftell(lInputFile);
01132         rewind(lInputFile);
01133 
01134         buffer = (char*) malloc (sizeof(char)*lSize + 1);    // allocate memory
01135         result = fread(buffer, 1, lSize, lInputFile);        // copy file into the buffer
01136         buffer[lSize] = '\0';
01137         
01138         fclose(lInputFile);
01139         
01140         std::string tmp(buffer);
01141         free(buffer);
01142 
01143         if((strstr(targetcode, ".txt") != NULL) || (strstr(targetcode, ".TXT") != NULL))
01144         {
01145             std::smatch match;
01146             // regex pattern for TI-txt (captures full memory blocks)
01147             std::regex regex("@(.+)\n([^qQ@]+)");
01148 
01149             std::regex_search(tmp, match, regex);
01150             std::string address = match[1];
01151             std::string code = match[2];
01152         
01153             code.erase(remove(code.begin(), code.end(), ' '), code.end());    //remove whitespaces
01154             code.erase(remove(code.begin(), code.end(), '\n'), code.end()); //and newlines
01155 
01156             char * p;
01157             *lStartAddress = (LONG)strtol(address.c_str(), &p, 16);
01158             *lProgramLength = (LONG)(code.size()/2);
01159         }
01160         else if((strstr(targetcode, ".a43") != NULL) || (strstr(targetcode, ".A43") != NULL) ||
01161                 (strstr(targetcode, ".hex") != NULL) || (strstr(targetcode, ".HEX") != NULL) ||
01162                 (strstr(targetcode, ".int") != NULL) || (strstr(targetcode, ".INT") != NULL))
01163         {
01164             std::smatch match;
01165             // regex pattern for INTEL-hex (captures data from one line only)
01166             std::regex regex(":(\\w\\w)(\\w\\w\\w\\w)(\\w\\w)(\\w*)\\w\\w");
01167         
01168             LONG next_address = 0;
01169             LONG bytecount = 0;
01170             LONG address = 0;
01171             LONG recordtype = 0;
01172             LONG prev_recordtype = 0;
01173             LONG length = 0;
01174 
01175             while(std::regex_search(tmp, match, regex))
01176             {
01177                 char * p;
01178                 bytecount = strtol(string(match[1]).c_str(), &p, 16);
01179                 address = strtol(string(match[2]).c_str(), &p, 16);
01180                 recordtype = strtol(string(match[3]).c_str(), &p, 16);
01181                 std::string data = match[4];
01182                 
01183                 length += bytecount;
01184 
01185                 if(next_address)
01186                 {
01187                     if((prev_recordtype == 2) || 
01188                        (prev_recordtype == 4) &&
01189                        (recordtype == 0))
01190                     {
01191                         *lStartAddress += address;
01192                     }
01193                     else if(address != next_address)
01194                     {
01195                         break;
01196                     }
01197                 }
01198                 else
01199                 {
01200                     switch(recordtype)
01201                     {
01202                         case 0:
01203                             *lStartAddress = address;
01204                         case 2:
01205                             *lStartAddress = address << 4;
01206                         case 4:
01207                             *lStartAddress = address << 16;
01208                     }
01209                 }
01210 
01211                 next_address = address + bytecount;
01212                 tmp.erase(0, bytecount + 12);
01213             }
01214 
01215             *lProgramLength = length;
01216         }
01217 
01218         return STATUS_OK;
01219     }
